--PROCEDURE QUE FILTRE A BUSCA DE UM FUNCIONARIO COM SEU REGISTRO COMPLETO--

CREATE PROCEDURE R_FUNCIONARIO(
               @NOME VARCHAR(40))
AS

                 SELECT IDFUNCIONARIO, NOME, GÊNERO, CARGO, SALÁRIO
                  FROM FUNCIONARIOS
                  WHERE NOME = @NOME
GO
